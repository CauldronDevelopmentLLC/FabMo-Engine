api {get} /file_commands/run/:id run a file (by its file id), apiGroup Files Commands, and apiSuccess Object} {status:"success",data : null

run = function(req, res, next) { File.get_by_id(req.params.id,function(file){ if(!file){res.send(404);return;} file.saverun();//update last run and run count information machine.runFile(file.path); var answer = { status:"success", data : null }; res.json(answer); }); }; module.exports = function(server) { server.get('/run/:id',run); //OK server.get('/quit',quit); //OK server.get('/pause',pause); //OK server.get('/resume',resume); //OK };
